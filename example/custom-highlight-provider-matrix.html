@dataclass

    name: str

    base = sum(values) or 1

  attr_reader :name, :score
  def initialize(name, score)
    @name = name
    @score = score
  end
end

users = [User.new("cat", 30), User.new("dog", 10), User.new("fox", 20)]
users.sort_by { |u| -u.score }.each do |u|
  puts "#{u.name}:#{u.score}"
